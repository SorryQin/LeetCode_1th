## 150.逆波兰表达式求值

这道题目没啥难度，唯一的问题就是如何将string类型的值转化为int型，我刚开始的写法有点问题：
```c++
        for (string s: tokens) {
            if (s != "+" && s != "-" && s != "*" && s != "/") {
                stk.push((int) s);
            }
```
在将字符串转换为整数时，直接对字符串使用强制类型转换 (int) s 是错误的。应该使用 std::stoi 或其他适当的方法来将字符串转换为整数

正确的解法应该是：
```c++
        for (string s : tokens) {
            if (s != "+" && s != "-" && s != "*" && s != "/") {
                // 使用 std::stoi 将字符串转换为整数
                stk.push(stoi(s));
            }
```
完整的代码如下：
```c++
class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        stack<int> stk;
        int tmp; // 存储运算结果
        for (string s: tokens) {
            if (s != "+" && s != "-" && s != "*" && s != "/") {
                stk.push(stoi(s));
            }
            else if (s == "+") {
                int i = stk.top();
                stk.pop();
                tmp = i + stk.top();
                stk.pop();
                stk.push(tmp);
            }
            else if (s == "-") {
                int i = stk.top();
                stk.pop();
                tmp = stk.top() - i;
                stk.pop();
                stk.push(tmp);
            }
            else if (s == "*") {
                int i = stk.top();
                stk.pop();
                tmp = i * stk.top();
                stk.pop();
                stk.push(tmp);
            }
            else if (s == "/") {
                int i = stk.top();
                stk.pop();
                tmp = stk.top() / i;
                stk.pop();
                stk.push(tmp);
            }            
        }
        return stk.top();
    }
};
```
ai给出了一种更为可读的代码，使用switch取代了复杂的for循环，代码如下：
```c++
#include <stack>
#include <vector>
#include <string>
using namespace std;

class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        stack<int> stk;
        for (string s : tokens) {
            if (s != "+" && s != "-" && s != "*" && s != "/") {
                // 使用 std::stoi 将字符串转换为整数
                stk.push(stoi(s));
            } else {
                int num2 = stk.top(); stk.pop();
                int num1 = stk.top(); stk.pop();
                switch (s.c_str()[0]) { // 使用 c_str() 获取字符串的第一个字符进行比较
                // 这里如果替换成：switch (s[0]) 也没问题
                    case '+':
                        stk.push(num1 + num2);
                        break;
                    case '-':
                        stk.push(num1 - num2);
                        break;
                    case '*':
                        stk.push(num1 * num2);
                        break;
                    case '/':
                        stk.push(num1 / num2);
                        break;
                }
            }
        }
        return stk.top();
    }
};
```
