## 零钱兑换II

典型的背包问题，二维数组的代码如下：

```c++
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        int bagSize = amount;

        vector<vector<uint64_t>> dp(coins.size(), vector<uint64_t>(bagSize + 1, 0));

        // 初始化最上行
        for (int j = 0; j <= bagSize; j++) {
            if (j % coins[0] == 0) dp[0][j] = 1;
        }
        // 初始化最左列
        for (int i = 0; i < coins.size(); i++) {
            dp[i][0] = 1;
        }
        // 以下遍历顺序行列可以颠倒
        for (int i = 1; i < coins.size(); i++) { // 行，遍历物品
            for (int j = 0; j <= bagSize; j++) { // 列，遍历背包
                if (coins[i] > j) dp[i][j] = dp[i - 1][j]; 
                else dp[i][j] = dp[i - 1][j] +  dp[i][j - coins[i]];
            }
        }
        return dp[coins.size() - 1][bagSize];
    }
};
```

一维数组的代码如下：

```c++
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        vector<uint64_t> dp(amount + 1, 0); // 防止相加数据超int
        dp[0] = 1; // 只有一种方式达到0
        for (int i = 0; i < coins.size(); i++) { // 遍历物品
            for (int j = coins[i]; j <= amount; j++) { // 遍历背包
                dp[j] += dp[j - coins[i]];
            }
        }
        return dp[amount]; // 返回组合数
    }
};
```

实现代码如下：

```c++
class Solution {
public:
    int combinationSum4(vector<int>& nums, int target) {
        sort(nums.begin(), nums.end());
        vector<int> dp(target + 1, 0);
        dp[0] = 1;
        for (int i = 1; i <= target; i++) {
            for (int j: nums) {
                if (j <= i && dp[i] < INT_MAX - dp[i - j]) {
                    dp[i] += dp[i - j];
                }
            }
        }
        return dp[target];
    }
};
```

## 爬楼梯（进阶版）

这也是个简单的背包问题，但是刚开始写的代码老是输出错误，改了半天也没找到问题之所在，后来才发现原来是 m 和 n 的值输入反了

实现代码如下：

```c++
#include <vector>
#include <iostream>

using namespace std;

int main(){
    // input
    // m = pace, n = stage
    int m, n;
    cin >> n >> m;

    vector<int> dp(n + 1, 0);
    dp[0] = 1;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            if (j <= i) {
                dp[i] += dp[i - j];
            }
        }
    }

    cout << dp[n] << endl;

    return 0;
}
```
